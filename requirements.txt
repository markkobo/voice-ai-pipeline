# Voice AI Pipeline Requirements
#
# INSTALLATION (recommended order for a fresh GPU pod):
#   sudo apt install -y ffmpeg sox cifs-utils
#   sudo apt install -y nvidia-driver-595 nvidia-utils-595   # for Ubuntu 24.04 + A10G/A100; verify with `nvidia-detector`
#   sudo modprobe nvidia                                      # or reboot
#   python3 -m venv .venv && source .venv/bin/activate
#   pip install torch==2.6.0 torchaudio==2.6.0 torchvision==0.21.0 \
#       --index-url https://download.pytorch.org/whl/cu124
#   pip install --use-deprecated=legacy-resolver -r requirements.txt
#
# WHY EACH PIN: every "==" below corresponds to a real version conflict found
# during install on Ubuntu 24.04. The legacy resolver alone resolves to a
# broken combo (torchvision 0.26 vs torch 2.6 → "torchvision::nms doesn't
# exist"; tokenizers 0.23.1 vs qwen-asr/tts cap; etc.). The pins are the
# minimum set that satisfies both pip's resolver and the qwen-* runtime
# version checks.

# PyTorch (must install separately first — see header)
# ============================================================================

# command uses --index-url https://download.pytorch.org/whl/cu124 so the
# +cu124 wheels are picked up.
torch==2.6.0

torchvision==0.21.0
# Why 0.21.0 (not >=0.20.0): the loose constraint resolved to 0.26.0 which is
# built against torch 2.9 and aborts with "operator torchvision::nms does not
# exist" on torch 2.6. 0.21.0 is the matching release for torch 2.6.
